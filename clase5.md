<?php
  if (isset($_POST['enviar']) && $_POST['enviar'] == '1') {
    // print_r($_POST);
    $nombre = $_POST['nombre'];
    $correo = $_POST['correo'];
    $pais = $_POST['pais'];
    if (empty($nombre)) {
      echo "Debe ingresar su nombre";
      exit;
    } if (empty($correo)) {
      echo "Debe ingresar su correo";
      exit;
    } if ($pais == 0) {
      echo "Debe seleccionar un país";
      exit;
    }
  }
 ?>
